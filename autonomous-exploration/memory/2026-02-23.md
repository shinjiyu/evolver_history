## 一、理论基础调研

### 1.1 目标分解的经典方法

#### GOAP (Goal-Oriented Action Planning)
- **原理**: 从目标状态反向推导，找到一系列动作达到目标
- **优势**: 灵活、能处理动态环境
- **在 LLM Agent 中的应用**:
  - 可以让 LLM 生成动作空间（Action Space）
  - 使用 LLM 评估动作的前置条件和效果
  - **可行性**: 高 - LLM 天然擅长理解和生成动作描述

#### HTN (Hierarchical Task Network)
- **原理**: 将高层任务递归分解为低层任务，直到可执行的原子任务
- **核心组件**:
  - 复合任务（Compound Task）: 需要进一步分解
  - 原子任务（Primitive Task）: 可直接执行
  - 方法（Method）: 分解规则
- **在 LLM Agent 中的应用**:
  - LLM 可以作为"方法选择器"
  - 动态生成任务分解策略
  - **可行性**: 中 - 需要设计良好的任务模板系统

#### 行为树 (Behavior Trees)
- **原理**: 树状结构控制决策流程，支持序列、选择、并行等节点
- **优势**: 模块化、可复用、易于调试
- **在 LLM Agent 中的应用**:
  - LLM 可以动态生成行为树节点
  - 可以根据上下文调整树结构
  - **可行性**: 中 - 需要定义清晰的节点类型

### 1.2 层次规划的最新进展

#### LLM + Planning 的融合
- **Self-Refine**: LLM 自我反思和修正计划
- **Tree of Thoughts**: 探索多个推理路径，选择最优
- **ReWOO**: 推理与观察解耦，提高计划效率

#### 关键发现
1. **计划生成**: LLM 可以生成合理的初步计划（零样本/少样本）
2. **计划评估**: LLM 可以评估计划的可行性和风险
3. **计划调整**: LLM 可以根据反馈动态调整计划

### 1.3 多目标优化策略

#### 常见方法
1. **加权求和法**: `score = w1*obj1 + w2*obj2 + ...`
2. **Pareto 优化**: 寻找非支配解集
3. **Lexicographic**: 按优先级顺序优化
4. **约束满足**: 将某些目标转为约束条件

#### 在 LLM Agent 中的应用
- **Prompt Engineering**: 在提示词中明确优先级
- **Self-Consistency**: 生成多个计划，选择综合最优
- **辩论机制**: 多个 Agent 代表不同目标，通过辩论达成平衡

### 2.2 目标分解算法

```javascript
/**
 * 目标分解器 - 将模糊目标分解为具体目标
 */
class GoalDecomposer {
  
  /**
   * 分解目标
   * @param {Goal} vagueGoal - 模糊目标
   * @param {UserModel} userModel - 用户模型
   * @param {Context} context - 当前上下文
   * @returns {GoalTree} 目标树
   */
  async decompose(vagueGoal, userModel, context) {
    const goalTree = {
      root: vagueGoal,
      children: []
    };
    
    // Step 1: 识别目标类型
    const goalType = await this.classifyGoal(vagueGoal);
    
    // Step 2: 应用分解策略
    switch (goalType) {
      case 'information_gathering':
        goalTree.children = await this.decomposeInfoGathering(vagueGoal);
        break;
      case 'task_execution':
        goalTree.children = await this.decomposeTaskExecution(vagueGoal);
        break;
      case 'learning':
        goalTree.children = await this.decomposeLearning(vagueGoal);
        break;
      default:
        goalTree.children = await this.decomposeGeneric(vagueGoal);
    }
    
    // Step 3: 验证子目标完整性
    await this.validateGoalTree(goalTree);
    
    // Step 4: 添加执行条件
    for (const child of goalTree.children) {
      child.preconditions = await this.inferPreconditions(child);
      child.priority = await this.calculatePriority(child, userModel);
    }
    
    return goalTree;
  }
  
  /**
   * 信息收集类目标分解
   */
  async decomposeInfoGathering(goal) {
    // 示例: "了解 RAG 最新进展" → 分解为多个搜索任务
    return [
      { 
        description: "搜索最近3个月的RAG相关论文",
        type: "search",
        params: { query: "RAG retrieval augmented generation 2024", timeRange: "3m" }
      },
      {
        description: "查找主流RAG框架的更新日志",
        type: "search",
        params: { sources: ["github.com", "huggingface.co"], tags: ["RAG", "framework"] }
      },
      {
        description: "收集RAG优化技巧的博客文章",
        type: "search",
        params: { sources: ["medium.com", "blog"], keywords: ["RAG optimization"] }
      },
      {
        description: "汇总并生成报告",
        type: "synthesis",
        dependsOn: [0, 1, 2]
      }
    ];
  }
  
  /**
   * 计算目标优先级
   */
  async calculatePriority(goal, userModel) {
    const factors = {
      // 用户相关性 (0-1)
      relevance: await this.assessRelevance(goal, userModel),
      
      // 紧急程度 (0-1)
      urgency: await this.assessUrgency(goal),
      
      // 预期价值 (0-1)
      value: await this.assessValue(goal, userModel),
      
      // 执行成本 (0-1, 1=低成本)
      feasibility: await this.assessFeasibility(goal)
    };
    
    // 加权求和
    return (
      factors.relevance * 0.35 +
      factors.urgency * 0.25 +
      factors.value * 0.25 +
      factors.feasibility * 0.15
    );
  }
}
```

### 2.3 目标冲突检测与解决

```javascript
/**
 * 目标冲突检测器
 */
class GoalConflictDetector {
  
  /**
   * 检测目标冲突
   * @param {Goal[]} goals - 待检测的目标列表
   * @returns {Conflict[]} 检测到的冲突
   */
  async detectConflicts(goals) {
    const conflicts = [];
    
    for (let i = 0; i < goals.length; i++) {
      for (let j = i + 1; j < goals.length; j++) {
        const conflict = await this.checkPair(goals[i], goals[j]);
        if (conflict) {
          conflicts.push(conflict);
        }
      }
    }
    
    return conflicts;
  }
  
  /**
   * 检查两个目标之间的冲突
   */
  async checkPair(goalA, goalB) {
    // 资源冲突
    if (this.resourceConflict(goalA, goalB)) {
      return {
        type: 'resource',
        goals: [goalA, goalB],
        resource: goalA.resource,
        resolution: 'sequential' // 串行执行
      };
    }
    
    // 时间冲突
    if (this.timeConflict(goalA, goalB)) {
      return {
        type: 'time',
        goals: [goalA, goalB],
        resolution: 'prioritize' // 按优先级执行
      };
    }
    
    // 逻辑冲突
    if (await this.logicalConflict(goalA, goalB)) {
      return {
        type: 'logical',
        goals: [goalA, goalB],
        description: '目标相互矛盾',
        resolution: 'user_choice' // 需要用户决策
      };
    }
    
    return null;
  }
  
  /**
   * 解决冲突
   */
  async resolveConflict(conflict, userModel) {
    switch (conflict.resolution) {
      case 'sequential':
        return {
          strategy: '串行执行',
          order: await this.determineOrder(conflict.goals, userModel)
        };
        
      case 'prioritize':
        return {
          strategy: '优先级排序',
          order: conflict.goals.sort((a, b) => b.priority - a.priority)
        };
        
      case 'user_choice':
        return {
          strategy: '用户决策',
          message: `检测到目标冲突：${conflict.description}。请选择执行哪个目标。`
        };
        
      default:
        return { strategy: '未知策略' };
    }
  }
}
```

### 2.4 长期目标与短期任务平衡

```javascript
/**
 * 目标平衡器 - 平衡长期目标和短期任务
 */
class GoalBalancer {
  
  constructor() {
    // 资源分配策略
    this.allocationStrategy = {
      longTerm: 0.3,    // 30% 资源用于长期目标
      shortTerm: 0.5,   // 50% 资源用于短期任务
      reactive: 0.2     // 20% 资源预留响应突发事件
    };
  }
  
  /**
   * 规划目标执行
   * @param {Goal[]} longTermGoals - 长期目标
   * @param {Task[]} shortTermTasks - 短期任务
   * @param {number} availableTime - 可用时间（分钟）
   * @returns {Schedule} 执行计划
   */
  async plan(longTermGoals, shortTermTasks, availableTime) {
    const schedule = {
      longTermSlots: [],
      shortTermSlots: [],
      reactiveBuffer: availableTime * this.allocationStrategy.reactive
    };
    
    const usableTime = availableTime - schedule.reactiveBuffer;
    
    // 分配长期目标时间
    const longTermTime = usableTime * (this.allocationStrategy.longTerm / 
      (this.allocationStrategy.longTerm + this.allocationStrategy.shortTerm));
    
    // 分配短期任务时间
    const shortTermTime = usableTime - longTermTime;
    
    // 选择要执行的长期目标进度
    schedule.longTermSlots = await this.selectLongTermProgress(
      longTermGoals, 
      longTermTime
    );
    
    // 选择要执行的短期任务
    schedule.shortTermSlots = await this.selectShortTermTasks(
      shortTermTasks, 
      shortTermTime
    );
    
    return schedule;
  }
  
  /**
   * 选择长期目标的下一步进展
   */
  async selectLongTermProgress(goals, availableTime) {
    const selected = [];
    let remainingTime = availableTime;
    
    // 按优先级和进度排序
    const sorted = goals.sort((a, b) => {
      // 优先级高且进度慢的优先
      const aScore = a.priority * (1 - a.progress);
      const bScore = b.priority * (1 - b.progress);
      return bScore - aScore;
    });
    
    for (const goal of sorted) {
      // 获取下一步行动
      const nextStep = await this.getNextStep(goal);
      
      if (nextStep.estimatedTime <= remainingTime) {
        selected.push({
          goal: goal,
          step: nextStep,
          reason: `推进 "${goal.description}" (${(goal.progress * 100).toFixed(0)}% 完成)`
        });
        remainingTime -= nextStep.estimatedTime;
      }
      
      if (remainingTime < 5) break; // 至少预留5分钟
    }
    
    return selected;
  }
  
  /**
   * 动态调整策略
   */
  async adjustStrategy(executionHistory) {
    // 分析执行历史
    const analysis = await this.analyzeExecution(executionHistory);
    
    // 如果长期目标进度过慢，增加资源分配
    if (analysis.longTermProgress < 0.3) {
      this.allocationStrategy.longTerm += 0.05;
      this.allocationStrategy.shortTerm -= 0.05;
    }
    
    // 如果经常被突发事件打断，增加缓冲
    if (analysis.interruptionRate > 0.3) {
      this.allocationStrategy.reactive += 0.05;
      this.allocationStrategy.shortTerm -= 0.05;
    }
    
    // 确保分配合理
    const total = Object.values(this.allocationStrategy).reduce((a, b) => a + b, 0);
    for (const key in this.allocationStrategy) {
      this.allocationStrategy[key] /= total;
    }
  }
}
```

```javascript
/**
 * 目标生成器 - 从多个来源生成候选目标
 */
class GoalGenerator {
  
  /**
   * 生成候选目标
   * @param {UserModel} userModel - 用户模型
   * @param {Context} context - 当前上下文
   * @returns {CandidateGoal[]} 候选目标列表
   */
  async generateCandidates(userModel, context) {
    const candidates = [];
    
    // 1. 从用户明确目标分解
    const explicitGoals = await this.fromExplicitGoals(userModel);
    candidates.push(...explicitGoals.map(g => ({
      ...g,
      source: 'explicit',
      weight: 1.0
    })));
    
    // 2. 从兴趣图谱推断
    const interestGoals = await this.fromInterestGraph(userModel);
    candidates.push(...interestGoals.map(g => ({
      ...g,
      source: 'interest',
      weight: 0.7
    })));
    
    // 3. 从异常检测发现
    const anomalyGoals = await this.fromAnomalies(userModel, context);
    candidates.push(...anomalyGoals.map(g => ({
      ...g,
      source: 'anomaly',
      weight: 0.8
    })));
    
    // 4. 从知识盲区发现（好奇心驱动）
    const curiosityGoals = await this.fromCuriosity(userModel);
    candidates.push(...curiosityGoals.map(g => ({
      ...g,
      source: 'curiosity',
      weight: 0.5
    })));
    
    // 5. 从日历事件预判
    const calendarGoals = await this.fromCalendar(context);
    candidates.push(...calendarGoals.map(g => ({
      ...g,
      source: 'calendar',
      weight: 0.9
    })));
    
    return candidates;
  }
  
  /**
   * 从兴趣图谱生成目标
   */
  async fromInterestGraph(userModel) {
    const goals = [];
    
    // 获取兴趣强度上升最快的领域
    const trendingInterests = userModel.interestGraph.getTrending(7); // 最近7天
    
    for (const interest of trendingInterests) {
      // 生成探索目标
      goals.push({
        description: `深入了解 ${interest.topic}`,
        type: 'learning',
        metadata: {
          interest: interest.topic,
          trend: interest.trend
        }
      });
    }
    
    return goals;
  }
  
  /**
   * 从日历事件预判目标
   */
  async fromCalendar(context) {
    const goals = [];
    const upcomingEvents = context.calendar.getUpcoming(24 * 60); // 未来24小时
    
    for (const event of upcomingEvents) {
      // 会议预研
      if (event.type === 'meeting') {
        goals.push({
          description: `为 "${event.title}" 准备资料`,
          type: 'preparation',
          deadline: event.startTime,
          metadata: {
            eventId: event.id,
            attendees: event.attendees
          }
        });
      }
      
      // 截止日期提醒
      if (event.type === 'deadline') {
        goals.push({
          description: `完成 "${event.title}"`,
          type: 'task',
          deadline: event.deadline,
          metadata: {
            taskId: event.id
          }
        });
      }
    }
    
    return goals;
  }
  
  /**
   * 好奇心驱动的目标生成
   */
  async fromCuriosity(userModel) {
    const goals = [];
    
    // 找到知识图谱中的空白区域
    const knowledgeGaps = await this.findKnowledgeGaps(userModel.knowledgeGraph);
    
    for (const gap of knowledgeGaps) {
      // 计算好奇心分数
      const curiosityScore = await this.calculateCuriosity(gap, userModel);
      
      if (curiosityScore > 0.6) {
        goals.push({
          description: `探索 ${gap.topic}`,
          type: 'exploration',
          metadata: {
            gap: gap,
            curiosityScore: curiosityScore
          }
        });
      }
    }
    
    return goals;
  }
}
```

### 4.1 核心类结构

```javascript
/**
 * 自主目标系统原型
 */
class AutonomousGoalSystem {
  constructor() {
    this.goalDecomposer = new GoalDecomposer();
    this.conflictDetector = new GoalConflictDetector();
    this.goalBalancer = new GoalBalancer();
    this.goalGenerator = new GoalGenerator();
    this.goalExecutor = new GoalExecutor();
    
    // 目标存储
    this.longTermGoals = [];
    this.activeGoals = [];
    this.completedGoals = [];
  }
  
  /**
   * 主循环
   */
  async runCycle(userModel, context) {
    // Step 1: 生成候选目标
    const candidates = await this.goalGenerator.generateCandidates(userModel, context);
    
    // Step 2: 评分和筛选
    const scoredGoals = await this.scoreAndFilter(candidates, userModel);
    
    // Step 3: 检测冲突
    const conflicts = await this.conflictDetector.detectConflicts(scoredGoals);
    
    // Step 4: 解决冲突
    const resolvedGoals = await this.resolveConflicts(scoredGoals, conflicts);
    
    // Step 5: 分解目标
    const decomposedGoals = [];
    for (const goal of resolvedGoals) {
      const goalTree = await this.goalDecomposer.decompose(goal, userModel, context);
      decomposedGoals.push(goalTree);
    }
    
    // Step 6: 平衡长期和短期
    const schedule = await this.goalBalancer.plan(
      this.longTermGoals,
      decomposedGoals,
      context.availableTime
    );
    
    // Step 7: 执行
    const results = await this.goalExecutor.execute(schedule);
    
    // Step 8: 学习
    await this.learn(results, userModel);
    
    return results;
  }
  
  /**
   * 从执行结果中学习
   */
  async learn(results, userModel) {
    for (const result of results) {
      // 成功的目标 → 强化相关模式
      if (result.success) {
        await userModel.reinforcePattern(result.goal);
      }
      
      // 失败的目标 → 分析原因并调整
      if (result.failure) {
        await userModel.analyzeFailure(result.goal, result.error);
      }
    }
    
    // 更新用户模型
    await userModel.update();
  }
}
```

## 五、关键发现总结

### 5.1 技术发现

1. **目标分解最适合 HTN 方法**
   - 原因: LLM 天然擅长理解层次结构
   - 实现难度: 中
   - 建议优先级: 高

2. **多目标优化应采用 Lexicographic + 加权混合**
   - 原因: 简单有效，易于理解和调整
   - 实现难度: 低
   - 建议优先级: 高

3. **冲突检测可简化为三类**
   - 资源冲突 → 串行执行
   - 时间冲突 → 优先级排序
   - 逻辑冲突 → 用户决策

### 5.2 创新点

1. **目标来源多元化**
   - 不止从用户指令，还从兴趣图谱、日历、异常、好奇心等多处生成

2. **动态资源分配**
   - 长期/短期/反应式三类资源动态调整

3. **学习闭环**
   - 从成功/失败中持续学习，优化目标生成策略
