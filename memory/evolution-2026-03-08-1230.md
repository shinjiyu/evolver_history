```bash
# safe_edit 函数
safe_edit() {
    local file=$1
    local old_text="$2"
    local new_text="$3"
    local max_retries=3
    local retry=0
    
    while [ $retry -lt $max_retries ]; do
        # 重新读取最新内容
        current_content=$(cat "$file")
        
        # 检查是否匹配
        if echo "$current_content" | grep -qF "$old_text"; then
            # 执行替换（原子性）
            temp_file="${file}.tmp.$$"
            echo "$current_content" | sed "s|$old_text|$new_text|" > "$temp_file"
            mv "$temp_file" "$file"
            echo "✅ Edit 成功"
            return 0
        fi
        
        retry=$((retry + 1))
        sleep 0.5
    done
    
    echo "❌ Edit 失败"
    return 1
}
```

```bash
# safe_read 函数
safe_read() {
    local file=$1
    local default_content="${2:-}"
    
    if [ ! -f "$file" ]; then
        echo "⚠️  文件不存在: $file"
        
        if [ -n "$default_content" ]; then
            mkdir -p "$(dirname "$file")"
            echo "$default_content" > "$file"
            echo "  ✅ 已创建默认文件"
            echo "$default_content"
        else
            return 1
        fi
    else
        cat "$file"
    fi
}
```
