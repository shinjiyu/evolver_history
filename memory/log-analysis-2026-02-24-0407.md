```javascript
const MAX_API_CALLS_PER_RUN = 5;
const RATE_LIMIT_BACKOFF_MS = 5 * 60 * 1000; // 5 分钟

let apiCallCount = 0;
async function safeApiCall(fn) {
  if (apiCallCount >= MAX_API_CALLS_PER_RUN) {
    console.log('Reached max API calls for this run');
    return null;
  }
  
  try {
    const result = await fn();
    apiCallCount++;
    return result;
  } catch (error) {
    if (error.status === 429) {
      console.log('Rate limit hit, backing off...');
      await sleep(RATE_LIMIT_BACKOFF_MS);
    }
    throw error;
  }
}
```

```javascript
async function fetchWithRetry(url, options = {}, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, {
        ...options,
        timeout: 30000 // 30 秒超时
      });
      
      if (response.ok) return response;
      
      // 5xx 错误重试
      if (response.status >= 500 && i < retries - 1) {
        await sleep(2000 * (i + 1)); // 指数退避
        continue;

      throw new Error(`HTTP ${response.status}`);

      await sleep(2000 * (i + 1));

}
```
